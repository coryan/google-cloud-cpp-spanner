99 template <typename Op>
100 class WriteMutationBuilder {
101  public:
102  WriteMutationBuilder(std::string table_name,
103  std::vector<std::string> column_names) {
104  auto& field = Op::mutable_field(m_);
105  field.set_table(std::move(table_name));
106  field.mutable_columns()->Reserve(column_names.size());
107  for (auto& name : column_names) {
108  *field.add_columns() = std::move(name);
109  }
110  }
111 
112  Mutation Build() const& { return Mutation(m_); }
113  Mutation Build() && { return Mutation(std::move(m_)); }
114 
115  template <typename... Ts>
116  WriteMutationBuilder& AddRow(Row<Ts...> row) {
117  auto& lv = *Op::mutable_field(m_).add_values();
118  for (auto& v : std::move(row).values()) {
119  std::tie(std::ignore, *lv.add_values()) = internal::ToProto(std::move(v));
120  }
121  return *this;
122  }
123 
124  template <typename... Ts>
125  WriteMutationBuilder& EmplaceRow(Ts&&... values) {
126  return AddRow(MakeRow(std::forward<Ts>(values)...));
127  }
128 
129  private:
130  google::spanner::v1::Mutation m_;
131 };
