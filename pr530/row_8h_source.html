284 // A helper metafunction that promots some C++ literal types to the types
285 // required by Cloud Spanner. For example, a literal 42 is of type int, but
286 // this will promote it to type std::int64_t. Similarly, a C++ string literal
287 // will be "promoted" to type std::string.
288 template <typename T>
289 T PromoteLiteralImpl(T);
290 std::string PromoteLiteralImpl(char const*);
291 std::int64_t PromoteLiteralImpl(int);
292 template <typename T>
293 using PromoteLiteral = decltype(PromoteLiteralImpl(std::declval<T>()));

295 // A helper functor to be used with `internal::ForEach` to iterate the columns
296 // of a `Row<Ts...>` in the ParseRow() function.
297 struct ExtractValue {
298  Status& status;
299  template <typename T, typename It>
300  void operator()(T& t, It& it) const {
301  auto x = it++->template get<T>();
302  if (!x) {
303  status = std::move(x).status();
304  } else {
305  t = *std::move(x);
306  }
307  }
308  template <typename It>
309  void operator()(Value& v, It& it) const {
310  v = *it++;
311  }

351 template <typename... Ts>
352 StatusOr<Row<internal::PromoteLiteral<Ts>...>> ParseRow(
353  std::array<Value, sizeof...(Ts)> const& array) {
354  Row<internal::PromoteLiteral<Ts>...> row;
355  auto it = array.begin();
356  Status status;
357  internal::ForEach(row, internal::ExtractValue{status}, it);
358  if (!status.ok()) return status;
