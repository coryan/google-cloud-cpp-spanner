27 std::string RandomDatabaseName(
28  google::cloud::internal::DefaultPRNG& generator) {
29  // A database ID must be between 2 and 30 characters, fitting the regular
30  // expression `[a-z][a-z0-9_\-]*[a-z0-9]`
31  int max_size = 30;
32  std::string const prefix = "db-";
33  auto size = static_cast<int>(max_size - 1 - prefix.size());
34  return prefix +
35  google::cloud::internal::Sample(
36  generator, size, "abcdefghijlkmnopqrstuvwxyz012345689_-") +
37  google::cloud::internal::Sample(generator, 1,
38  "abcdefghijlkmnopqrstuvwxyz");
39 }

41 void CreateDatabase(std::vector<std::string> const& argv) {
42  if (argv.size() != 3) {
43  throw std::runtime_error(
44  "create-database <project-id> <instance-id> <database-id>");
45  }
46 
48  using google::cloud::future;
49  using google::cloud::StatusOr;
50  [](std::string const& project_id, std::string const& instance_id,
51  std::string const& database_id) {
52  google::cloud::spanner::DatabaseAdminClient client;
53  google::cloud::spanner::Database database(project_id, instance_id,
54  database_id);
55  future<StatusOr<google::spanner::admin::database::v1::Database>> future =
56  client.CreateDatabase(database, {R"""(
57  CREATE TABLE Singers (
58  SingerId INT64 NOT NULL,
59  FirstName STRING(1024),
60  LastName STRING(1024),
61  SingerInfo BYTES(MAX)
62  ) PRIMARY KEY (SingerId))""",
63  R"""(
64  CREATE TABLE Albums (
65  SingerId INT64 NOT NULL,
66  AlbumId INT64 NOT NULL,
67  AlbumTitle STRING(MAX)
68  ) PRIMARY KEY (SingerId, AlbumId),
69  INTERLEAVE IN PARENT Singers ON DELETE CASCADE)"""});
70  StatusOr<google::spanner::admin::database::v1::Database> db = future.get();
71  if (!db) {
72  throw std::runtime_error(db.status().message());
73  }
74  std::cout << "Created database [" << database << "]\n";
75  }
77  (argv[0], argv[1], argv[2]);
78 }

215 void DeleteData(google::cloud::spanner::Client client) {
216  namespace spanner = google::cloud::spanner;
217 
218  // Delete each of the albums by individual key, then delete all the singers
219  // using a key range.
220  auto delete_albums =
221  spanner::DeleteMutationBuilder(
222  "Albums",
223  spanner::KeySetBuilder<spanner::Row<std::int64_t, std::int64_t>>()
224  .Add(spanner::MakeRow(1, 1))
225  .Add(spanner::MakeRow(1, 2))
226  .Add(spanner::MakeRow(2, 1))
227  .Add(spanner::MakeRow(2, 2))
228  .Add(spanner::MakeRow(2, 3))
229  .Build())
230  .Build();
231  using SingersKey = spanner::Row<std::int64_t>;
232  auto delete_singers =
233  spanner::DeleteMutationBuilder(
234  "Singers", spanner::KeySetBuilder<SingersKey>(
235  MakeKeyRangeClosed(SingersKey(1), SingersKey(5)))
236  .Build())
237  .Build();
238 
239  auto commit_result = client.Commit(spanner::MakeReadWriteTransaction(),
240  {delete_albums, delete_singers});
241  if (!commit_result) {
242  throw std::runtime_error(commit_result.status().message());
243  }
244  std::cout << "Delete was successful [spanner_delete_data]\n";

249 void ReadOnlyTransaction(google::cloud::spanner::Client client) {
250  namespace spanner = google::cloud::spanner;
251  auto read_only = spanner::MakeReadOnlyTransaction();
252 
253  spanner::SqlStatement select(
254  "SELECT SingerId, AlbumId, AlbumTitle FROM Albums");
255 
256  // Read#1.
257  auto read1 = client.ExecuteSql(read_only, select);
258  std::cout << "Read 1 results\n";
259  for (auto row : read1->Rows<std::int64_t, std::int64_t, std::string>()) {
260  if (!row) {
261  throw std::runtime_error(row.status().message());
262  }
263  std::cout << "SingerId: " << row->get<0>() << " AlbumId: " << row->get<1>()
264  << " AlbumTitle: " << row->get<2>() << "\n";
265  }
266  // Read#2. Even if changes occur in-between the reads the transaction ensures
267  // that Read #1 and Read #2 return the same data.
268  auto read2 = client.ExecuteSql(read_only, select);
269  std::cout << "Read 2 results\n";
270  for (auto row : read2->Rows<std::int64_t, std::int64_t, std::string>()) {
271  if (!row) {
272  throw std::runtime_error(row.status().message());
273  }
274  std::cout << "SingerId: " << row->get<0>() << " AlbumId: " << row->get<1>()
275  << " AlbumTitle: " << row->get<2>() << "\n";
276  }
277 }

281 void ReadWriteTransaction(google::cloud::spanner::Client client) {
282  namespace spanner = google::cloud::spanner;
283  auto txn = spanner::MakeReadWriteTransaction();
284 
285  auto get_current_budget = [&txn, &client](std::int64_t singer_id,
286  std::int64_t album_id) {
287  auto key = spanner::KeySetBuilder<spanner::Row<std::int64_t, std::int64_t>>(
288  spanner::MakeRow(singer_id, album_id))
289  .Build();
290  auto read = client.Read(txn, "Albums", std::move(key), {"MarketingBudget"});
291  if (!read) {
292  throw std::runtime_error(read.status().message());
293  }
294  for (auto row : read->Rows<std::int64_t>()) {
295  if (!row) {
296  throw std::runtime_error(read.status().message());
297  }
298  // We expect at most one result from the `Read()` request. Return the
299  // first one.
300  return row->get<0>();
301  }
302  throw std::runtime_error("Key not found (" + std::to_string(singer_id) +
303  "," + std::to_string(album_id) + ")");
304  };
305 
306  auto b1 = get_current_budget(1, 1);
307  auto b2 = get_current_budget(2, 2);
308  std::int64_t transfer_amount = 200000;
309 
310  auto commit_result = client.Commit(
311  txn, {spanner::UpdateMutationBuilder(
312  "Albums", {"SingerId", "AlbumId", "MarketingBudget"})
313  .EmplaceRow(1, 1, b1 + transfer_amount)
314  .EmplaceRow(2, 2, b2 - transfer_amount)
315  .Build()});
316 
317  if (!commit_result) {
318  throw std::runtime_error(commit_result.status().message());
319  }
320  std::cout << "Transfer was successful [spanner_read_write_transaction]\n";
321 }

438  // Cloud Spanner STRUCT<> types with named fields are represented by

445  std::make_pair("LastName", std::move(last_name)));

448  make_name("Elena", "Campbell"),
449  make_name("Gabriel", "Wright"),
450  make_name("Benjamin", "Martinez"),

510  {"SongName", "Imagination"},
511  {"ArtistNames",

530 int RunOneCommand(std::vector<std::string> argv) {
531  using CommandType = std::function<void(std::vector<std::string> const&)>;
532 
533  using SampleFunction = void (*)(google::cloud::spanner::Client);
534 
535  using CommandMap = std::map<std::string, CommandType>;
536  auto make_command_entry = [](char const* sample_name, SampleFunction sample) {
537  auto make_command = [](char const* sample_name, SampleFunction sample) {
538  return [sample_name, sample](std::vector<std::string> const& argv) {
539  if (argv.size() != 3) {
540  throw std::runtime_error(std::string(sample_name) +
541  " <project-id> <instance-id> <database-id>");
542  }
543  sample(MakeSampleClient(argv[0], argv[1], argv[2]));
544  };
545  };
546  return CommandMap::value_type(sample_name,
547  make_command(sample_name, sample));
548  };
549 
550  CommandMap commands = {
551  {"create-database", &CreateDatabase},
552  {"add-column", &AddColumn},
553  {"drop-database", &DropDatabase},
554  {"quickstart", &QuickstartCommand},
555  make_command_entry("insert-data", &InsertData),
556  make_command_entry("update-data", &UpdateData),
557  make_command_entry("delete-data", &DeleteData),
558  make_command_entry("read-only-transaction", &ReadOnlyTransaction),
559  make_command_entry("read-write-transaction", &ReadWriteTransaction),
560  make_command_entry("dml-standard-insert", &DmlStandardInsert),
561  make_command_entry("dml-standard-update", &DmlStandardUpdate),
562  make_command_entry("dml-standard-delete", &DmlStandardDelete),
563  make_command_entry("write-data-for-struct-queries",
564  &WriteDataForStructQueries),
565  make_command_entry("query-data-with-struct", &QueryDataWithStruct),
566  make_command_entry("query-data-with-array-of-struct",
567  &QueryDataWithArrayOfStruct),
568  make_command_entry("field-access-struct-parameters",
569  &FieldAccessOnStructParameters),
570  make_command_entry("field-access-on-nested-struct",
571  &FieldAccessOnNestedStruct),
572  };
573 
574  static std::string usage_msg = [&argv, &commands] {
575  auto last_slash = std::string(argv[0]).find_last_of("/\\");
576  auto program = argv[0].substr(last_slash + 1);
577  std::string usage;
578  usage += "Usage: ";
579  usage += program;
580  usage += " <command> [arguments]\n\n";
581  usage += "Commands:\n";
582  for (auto&& kv : commands) {
583  try {
584  kv.second({});
585  } catch (std::exception const& ex) {
586  usage += " ";
587  usage += ex.what();
588  usage += "\n";
589  }
590  }
591  return usage;
592  }();
593 
594  if (argv.size() < 2) {
595  std::cerr << "Missing command argument\n" << usage_msg << "\n";
596  return 1;
597  }
598  std::string command_name = argv[1];
599  argv.erase(argv.begin()); // remove the program name from the list.
600  argv.erase(argv.begin()); // remove the command name from the list.
601 
602  auto command = commands.find(command_name);
603  if (commands.end() == command) {
604  std::cerr << "Unknown command " << command_name << "\n"
605  << usage_msg << "\n";
606  return 1;
607  }
608 
609  // Run the command.
