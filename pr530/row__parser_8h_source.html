166  // Consumes Values from value_source_ and stores the consumed Row in curr_.
167  // Called by iterator::operator++().
168  void Advance() {
169  if (curr_ && !*curr_) { // Last row was an error; jump to end
170  value_source_ = nullptr;
171  return;
172  }
173  std::array<Value, RowType::size()> values;
174  for (std::size_t i = 0; i < values.size(); ++i) {
175  StatusOr<optional<Value>> v = value_source_();
176  if (!v) {
177  curr_ = std::move(v).status();
178  return;
179  }
180  if (!*v) {
181  if (i == 0) { // We've successfully reached the end.
182  value_source_ = nullptr;
183  } else {
184  curr_ = Status(StatusCode::kUnknown, "incomplete row");
185  }
186  return;
187  }
188  values[i] = **std::move(v);
189  }
190  curr_ = ParseRow<T, Ts...>(values);
191  }

193  ValueSource value_source_; // nullpr is end
194  optional<value_type> curr_;
