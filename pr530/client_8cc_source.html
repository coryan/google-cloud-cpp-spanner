121 StatusOr<CommitResult> RunTransactionImpl(
122  Client& client, Transaction::ReadWriteOptions const& opts,
123  std::function<StatusOr<Mutations>(Client, Transaction)> const& f) {
124  Transaction txn = MakeReadWriteTransaction(opts);
125  StatusOr<Mutations> mutations;
126 #if GOOGLE_CLOUD_CPP_HAVE_EXCEPTIONS
127  try {
128 #endif
129  mutations = f(client, txn);
130 #if GOOGLE_CLOUD_CPP_HAVE_EXCEPTIONS
131  } catch (...) {
132  auto status = client.Rollback(txn);
133  if (!status.ok()) {
134  GCP_LOG(WARNING) << "Rollback() failure in RunTransaction(): "
135  << status.message();
136  }
137  throw;
138  }
139 #endif
140  if (!mutations) {
141  auto status = client.Rollback(txn);
142  if (!status.ok()) {
143  GCP_LOG(WARNING) << "Rollback() failure in RunTransaction(): "
144  << status.message();
145  }
146  return mutations.status();
147  }
148  return client.Commit(txn, *mutations);
149 }

155 StatusOr<CommitResult> RunTransactionWithPolicies(
156  Client client, Transaction::ReadWriteOptions const& opts,
157  std::function<StatusOr<Mutations>(Client, Transaction)> const& f,
158  std::unique_ptr<RetryPolicy> retry_policy,
159  std::unique_ptr<BackoffPolicy> backoff_policy) {
160  Status last_status(
161  StatusCode::kFailedPrecondition,
162  "Retry policy should not be exhausted when retry loop starts");
163  char const* reason = "Too many failures in ";
164  while (!retry_policy->IsExhausted()) {
165  auto result = RunTransactionImpl(client, opts, f);
166  if (result) return result;
167  last_status = std::move(result).status();
168  if (!retry_policy->OnFailure(last_status)) {
169  if (internal::SafeGrpcRetry::IsPermanentFailure(last_status)) {
170  reason = "Permanent failure in ";
171  }
172  break;
173  }
174  std::this_thread::sleep_for(backoff_policy->OnCompletion());
175  }
176  return internal::RetryLoopError(reason, __func__, last_status);
177 }
